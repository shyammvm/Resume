% Use a clean and professional document class for resumes:
\documentclass[a4paper,11pt]{article}

% --- Packages ---
\usepackage[utf8]{inputenc}
\usepackage[T1]{fontenc}              % Better font encoding
\usepackage{lmodern}                  % Clean, professional font
\usepackage{geometry}                 
\geometry{top=0.75in, bottom=1in, left=1in, right=1in}               
\usepackage{hyperref}                 
\hypersetup{
  colorlinks=true,
  urlcolor=blue,
  linkcolor=black,
  pdfborder={0 0 0}
}
\usepackage{enumitem}                 
\usepackage{fancyhdr}                 
\usepackage{setspace}                 % To control line spacing
\usepackage{parskip}                  % Space between paragraphs instead of indent

% --- Paragraph and spacing ---
\setlength{\parindent}{0pt}           % No paragraph indent
\setlength{\parskip}{6pt}             % Space between paragraphs
\setstretch{1.1}                      % Slightly more readable line spacing

% --- Header and footer ---
\fancypagestyle{plain}{%
  \fancyhf{}
  \renewcommand{\headrulewidth}{0pt}
  \renewcommand{\footrulewidth}{0pt}
}

% --- Custom commands ---
\newcommand{\name}[1]{%
  {\fontsize{18pt}{20pt}\selectfont\textbf{#1}} % Larger name font for emphasis
  \par\vspace{6pt}
}

\newcommand{\sectionTitle}[1]{%
  \vspace{12pt}
  \textbf{\large #1}
  \vspace{6pt}
  \par\nopagebreak
}

% --- Section formatting ---
\usepackage{titlesec}
% Clean section headings with smaller spacing and no numbers
\titleformat{\section}{\bfseries\large}{}{0em}{}
\titlespacing{\section}{0pt}{12pt}{6pt}

% --- Lists ---
\setlist[itemize]{noitemsep, topsep=2pt, leftmargin=1.5em}
